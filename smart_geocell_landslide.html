Arduino C++ arduino_main.ino — upload this to your board
// Smart Geocell Landslide Monitoring System
// Board: Arduino Uno / ESP32
// Author: ECE 249 CA2 – Roll 39,40,42,60

#include <SoftwareSerial.h>

// ── Pin definitions ──────────────────────
#define SOIL_PIN   A0
#define WATER_PIN  A1
#define RED_LED     7
#define GREEN_LED   6
#define BUZZER      8
#define SIM_TX     10
#define SIM_RX     11

// ── Thresholds (tune after calibration) ──
#define SOIL_THRESHOLD   600
#define WATER_THRESHOLD  450

// ── Alert phone number ────────────────────
String phoneNumber = "+91XXXXXXXXXX";

SoftwareSerial sim800(SIM_TX, SIM_RX);

bool alertSent = false;

void setup() {
  Serial.begin(9600);
  sim800.begin(9600);

  pinMode(RED_LED,   OUTPUT);
  pinMode(GREEN_LED, OUTPUT);
  pinMode(BUZZER,    OUTPUT);

  digitalWrite(GREEN_LED, HIGH);
  Serial.println("System started. Sensors active.");
  delay(2000);
}

void loop() {
  int soilVal  = analogRead(SOIL_PIN);
  int waterVal = analogRead(WATER_PIN);

  Serial.print("SOIL:");   Serial.print(soilVal);
  Serial.print(" WATER:"); Serial.println(waterVal);

  if (soilVal > SOIL_THRESHOLD || waterVal > WATER_THRESHOLD) {
    setDanger();
    if (!alertSent) {
      sendSMS("ALERT: Landslide risk detected! Soil/Water critical.");
      alertSent = true;
    }
  } else {
    setSafe();
    alertSent = false;
  }

  delay(3000);
}

void setDanger() {
  digitalWrite(GREEN_LED, LOW);
  digitalWrite(RED_LED,   HIGH);
  digitalWrite(BUZZER,    HIGH);
  Serial.println("STATUS: DANGER");
}

void setSafe() {
  digitalWrite(RED_LED,   LOW);
  digitalWrite(BUZZER,    LOW);
  digitalWrite(GREEN_LED, HIGH);
  Serial.println("STATUS: SAFE");
}

void sendSMS(String msg) {
  sim800.println("AT");          delay(500);
  sim800.println("AT+CMGF=1"); delay(500);
  sim800.print("AT+CMGS=\"");
  sim800.print(phoneNumber);
  sim800.println("\"");         delay(500);
  sim800.print(msg);
  sim800.write(26);            // Ctrl+Z sends the SMS
  delay(3000);
  Serial.println("SMS sent!");
}
Arduino C++ (ESP32 WiFi) esp32_wifi.ino — use this if you have ESP32
// ESP32 WiFi version — sends data to web dashboard
#include <WiFi.h>
#include <HTTPClient.h>

const char* ssid     = "YOUR_WIFI_NAME";
const char* password = "YOUR_WIFI_PASS";
const char* serverURL = "http://your-server.com/update";

#define SOIL_PIN  34
#define WATER_PIN 35

void setup() {
  Serial.begin(115200);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi connected: " + WiFi.localIP().toString());
}

void loop() {
  int soil  = analogRead(SOIL_PIN);
  int water = analogRead(WATER_PIN);

  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    String url = String(serverURL) +
                 "?soil="  + soil +
                 "&water=" + water;
    http.begin(url);
    int code = http.GET();
    Serial.println("HTTP: " + String(code));
    http.end();
  }
  delay(5000);
}
Python server.py — backend server to receive ESP32 data
# Install: pip install flask
# Run:     python server.py
# ESP32 will POST data to http://YOUR_IP:5000/update

from flask import Flask, request, jsonify
import json, datetime

app = Flask(__name__)
readings = []
SOIL_LIMIT  = 600
WATER_LIMIT = 450

@app.route('/update')
def update():
    soil  = int(request.args.get('soil',  0))
    water = int(request.args.get('water', 0))
    danger = soil > SOIL_LIMIT or water > WATER_LIMIT
    entry = {
        'time' : datetime.datetime.now().isoformat(),
        'soil' : soil,
        'water': water,
        'danger': danger
    }
    readings.append(entry)
    if len(readings) > 50: readings.pop(0)
    return jsonify({'status': 'danger' if danger else 'safe'})

@app.route('/data')
def data():
    return jsonify(readings[-20:])

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000, debug=True)

Important warnings

  • SIM800L requires 3.7–4.2V and up to 2A current — use a separate Li-ion battery or voltage regulator
  • Never power SIM800L directly from Arduino's 5V pin — it will reset or burn
  • Connect all GNDs together (Arduino GND + external battery GND)
  • Use waterproof tape or silicone on sensor connections in soil
